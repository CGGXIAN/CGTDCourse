==============================
Houdini VEX：数学函数
==============================

- int abs(int n) // 返回参数的绝对值
- float cos(float n) // cos函数
- float ceil(float n) // 向下求整，返回大于或等于参数的最小整数
- vector cross(vector a, vector b) // 叉乘
- float dot(vector a, vector b) // 点乘
- float floor(float n) // 向上求整，返回小于或等于参数的最大整数
- float frac(float n) // 返回参数的小数部分
- float length(vector v) // 返回矢量的长度
- <vector> normalize(<vector>v) // 矢量规格化
- float pow(float n, float exponent) // 次方
- float sin(float n) // sin函数
- float tan(float n) // tan函数
